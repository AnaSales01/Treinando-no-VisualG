var

HT,VH,PD,SB,TD,SL: real

inicio

escreval("Digite o valor da hora aula: ")
leia(VH)
escreval("Digite as horas trabalhadas no mês: ")
leia(HT)
escreval("Digite o percentual de desconto do INSS: ")
leia(PD)
SB <- HT * VH
escreval("O salário bruto é: ",SB)
TD <- PD/100 * SB
escreval("O total de deconto é: ",TD)
SL <- SB - TD
escreval("O salário líquido é: ",SL)
